\section*{Tropical Algebra}

\subsection*{A. Tropical Convexity}

\begin{itemize}
	\item \( S \subset \mathbb{R}^n \) is \textbf{\textit{tropically convex}} if \( x,y \in S \) and \( a,b \in \mathbb{R} \) implies that \( a \odot x \oplus b \odot y \in S \)
\end{itemize}
